\section{Patterns}
\label{patterns}

\def\ceq#1#2#3{\noindent\parbox[t]{25ex}{$\displaystyle #1$}\parbox{6ex}{\hfil $#2$}{$\displaystyle #3$}}

In the following $\U$ is some monster model of a complete theory $T$ of language $L$.
For simplicity $T$ is assumed to be one-sorted.

We define a structure with domain $\U^x$, where $x$ is some fixed tuple of variables. 
The language is denoted by $L^{\rm\! p}$.
Though $L^{\rm\! p}$ depends on $x$, this will be left implicit.

Let $V$ be some large set of variables that are copies of $x$. 
Let $\Delta$ be the set of positive Boolean combinations of formulas in $\bigcup_{x'\in V} L_{x',z}$, where $z=\langle z_i:i\in\omega\rangle$ is a fixed tuple of variables of length $\omega$.
For every $\phi(x_0,\dots,x_{n-1}\,;z)\in\Delta$ and every type $\alpha(z)\subseteq L$ the language $L^{\rm\! p}$ contains the $n$-ary relation \emph{$r_{\phi\,;\alpha}$.}
We require that $\alpha(z)$ is a finitary type, i.e.\@ only a finite number of variables actually occur in it.

Below by the symbol \emph{$L^{\rm\! p}$} we will also denote the set of \textit{atomic\/} formulas $r_{\phi\,;\alpha}(x_0,\dots,x_{n-1})$ for any $x_0,\dots,x_{n-1}\in V$.
Note that equalities are \textit{not\/} included in $L^{\rm\! p}$.

For a given model $M$, we define the interpretation of $r_{\phi\,;\alpha}$ in $\U^x$.
If $\bar a=a_0,\dots,a_{n-1}\in\U^x$, we declare $r_{\phi\,;\alpha}(\bar a)$ true if there is no $b\in M^z$ such that $\alpha(b)\ \wedge\ \phi(\bar a\,;b)$.
We repeat the definition for emphasis

\ceq{\hfill r_{\phi\,;\alpha}(\bar a)}{\Leftrightarrow}{\neg\exists z\in\alpha(M^z)\ \ \phi(\bar a\,;z)}

\ceq{}{\Leftrightarrow}{\phantom{\neg}\forall z\in\alpha(M^z)\ \neg\phi(\bar a\,;z).}

It is paramount to observe that the truth of $r_{\phi\,;\alpha}(\bar a)$ only depends on the equivalence class of $\bar a$ modulo $\equiv_M$.
Therefore it would be natural to define the interpretation of $L^{\rm\! p}$ on the quotient $\U^x/{\equiv_M}$ or in the type space $S_x(M)$, but we prefer to keep the original domain $\U^x$ to have a common playground with $L$.

A \emph{p-type\/} is a set of atomic formulas in $L^{\rm\! p}(\U^x)$.

Note incidentally that the relation $x_0\equiv_M x_1$ is definable by the p-type containing the formulas $r_{\theta\,; \top }(x_0,x_1)$ for every $\theta(x_0,x_1\,;z)$ of the form $\phi(x_0\,;z)\nleftrightarrow\phi(x_1\,;z)$.

We write \emph{$\textrm{p-tp}(\bar a)$} for the p-type of $\bar a$, i.e.\@ the set of formulas $r_{\phi\,;\alpha}(\bar x)$ such that $r_{\phi\,;\alpha}(\bar a)$ holds.

\begin{fact}\label{fact_p_qe}
  Formulas in $L^{\rm\! p}$ are closed under disjunction, conjunction, and existential quantification.
\end{fact}

\begin{proof}
  Note that for any two p-formulas $r_{\phi\,;\alpha}$ and $r_{\psi\,;\beta}$ we can always assume that $\alpha=\beta$.
  In fact, if we rename the variables that occur in $\beta(z)$ so that they are disjoint from those in $\alpha(z)$, we can replace both $\alpha(z)$ and $\beta(z)$ by $\alpha(z)\cup\beta(z)$.

  Now note that $r_{\phi\,;\alpha}\vee r_{\psi\,;\alpha}$ is equivalent to $r_{\phi\wedge\psi\,;\alpha}$ and $r_{\phi\,;\alpha}\wedge r_{\psi\,;\alpha}$ is equivalent to $r_{\phi\vee\psi\,;\alpha}$.

  Finally, consider the formula $\exists y\,r_{\phi\,;\alpha}(\bar x,y)$
  Without loss of generality, we can assume that $\phi$ has the form

  \ceq{\hfill\phi(\bar x,y\,;z)}{=}{\bigwedge_{i=1}^m\phi_i(\bar x,y\,;z)}

  where $\phi_i(\bar x,y;z)\,=\,\phi'_i(\bar x\,;z)\,\vee\,\psi_i(y\,;z)$ and each $\phi'_i(\bar x\,;z)$ is a disjunction of formulas in $\bigcup_{x'\in V} L_{x',z}$.
  Closure under existential quantification follows from the following equivalences

  \ceq{\hfill\exists y\ r_{\phi\,;\alpha}(\bar a, y)}{\Leftrightarrow}{\exists y\,\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^m\neg\phi_i(\bar a,y\,;z)\Big]}

  \ceq{}{\Leftrightarrow}{\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^m\neg\phi'_i(\bar a\,;z)\wedge\neg\forall y\,\psi_i(y\,;z)\Big]}

  % \ceq{}{\Leftrightarrow}{\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^n\neg\phi_i(a_i\,;z)

  % \ceq{}{\Leftrightarrow}{\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^n\neg\phi_i(a_i\,;z)\vee\neg\forall y\,\psi(y,z)\Big]}

  % \ceq{}{\Leftrightarrow}{\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^m\neg\phi''_i(\bar a\,;z)\Big]}

  \ceq{}{\Leftrightarrow}{\vphantom{\bigvee_{i=1}^n}r_{\theta,\beta}(\bar a),}

  where 
  
  \ceq{\hfill\theta(\bar x\,;z)}{=}{\bigwedge_{i=1}^m\phi'_i(\bar x\,;z)\ \vee\ \forall y\,\psi_i(y\,;z)}

  is a positive Boolean combination of formulas in  $\bigcup_{x'\in V} L_{x',z}$, as required.
\end{proof}


\begin{fact}[ ($\infty$-p-saturation)]\label{fact_psat}
  Let $\A\subseteq\U^x$ have arbitrary cardinality.
  Let $\bar x$ be a tuple of copies of $x$ of length $\le|\U|$.
  Then every p-type $p(\bar x)\subseteq L^{\rm\! p}(\A)$ that is finitely consistent in $\U^x$ is realized in $\U^x$.
\end{fact}

\begin{proof}

  Replacing $\A$ in $p(\bar x)$ by elements $A$ we obtain a equivalent type $p'(\bar x)\subseteq L^{\rm\! p}(A)$.
  Note that every formula $r_{\phi\,;\alpha}(\bar x)$ there is a type $q(\bar x)\subseteq L$ such that

  \ceq{\hfill\U^x\models r_{\phi\,;\alpha}(\bar a)}{\Leftrightarrow}{\U\models q(\bar a)}

  Therefore the saturation of $\U$ implies that $p'(\bar x)$ is realized in $\U^x$.

\section{Morphism}

A possible approach is to work in the quotient $\U^x/{\equiv_M}$.
A second possibility is to replace functions with relations.
The two options differs only in the notation.
Here we go for the second one that, arguably, simplifies the bookkeeping.

Let $R\mathrel{\subseteq}\U^x\times\U^x$ be a binary relation.
If $\bar a=\langle a_i:i<\lambda\rangle$ and $\bar b=\langle b_i:i<\lambda\rangle$ we write $\langle \bar a,\bar b\rangle\in R$ to abbreviate: $\langle a_i,b_i\rangle\in R$ for all $i<\lambda$.
% If we allow repetitions in $\bar a$ and $\bar b$ then every relation can be represented as $\{\langle\bar a,\bar b\rangle\}$.
% But note that $\langle\bar a',\bar b'\rangle\in\{\langle\bar a,\bar b\rangle\}$ does not imply that $\bar a'$ and $\bar b'$ are subtuples of $\bar a$ respectively $\bar b$ (because of the repetitions in $\bar a'$ and $\bar b'$).

We say that $R$ is a p-morphism if the following condition holds 

\ceq{\#\hfill\U^x\models r_{\phi\,;\alpha}(\bar a)}{\Rightarrow}{\U^x\models r_{\phi\,;\alpha}(\bar b)}\hfill for every $r_{\phi\,;\alpha}\in L^{\rm p}$ and every $\langle \bar a,\bar b\rangle\in R$.

When the converse implication in \# also holds, i.e.\@ when also $R^{-1}$ is a p-morphism, we say that $R$ is a \emph{partial p-isomorphism.}
When $\A\subseteq{\rm dom}\,R$,  $\B\subseteq{\rm rng}\,R$, and $R\cap (\A\times\B)$ is a partial p-isomorphism , we say that $R$ is an \emph{p-isomorphism} between $\A$ and $\B$ or, when $\A=\B$, a \emph{p-automorphism} of $\A$.

It is clear that if $R$ satisfiers \# then also  $\mathrel{\equiv_M}\circ \mathrel{R}\circ\mathrel{\equiv_M}$.
We say that the p-morphism $R$ is \emph{reduced\/} if there is no $R'\ \subset\ R$ such that $R\ \subseteq\ \big(\mathrel{\equiv_M}\circ \mathrel{R'}\circ\mathrel{\equiv_M}\big)$.

\begin{fact}
  For every p-morphism $R$ there is a reduced p-morphism $R'$ such that 
  
  \ceq{\flat\hfill\big(\mathrel{\equiv_M}\circ \mathrel{R}\circ\mathrel{\equiv_M}\big)}{=}{\big(\mathrel{\equiv_M}\circ \mathrel{R'}\circ\mathrel{\equiv_M}\big).}
\end{fact}

\begin{proof}
  Let $A\subseteq\U^x$ be a set of representatives of the equivalence classes of $\equiv_M$.
  Let $R'=\ \big(\mathrel{\equiv_M}\circ \mathrel{R}\circ\mathrel{\equiv_M}\big)\ \cap\ A\times A$.
  Clearly $R'$ satifies \flat.
  To prove that $R'$ is reduced, assume for a contradiction that $R''\ \subset\ R'$ also satisfies \flat.
  Let $\langle a,b\rangle\in R'\smallsetminus R''$.
  Then there are $a\equiv_M a'$ and $b\equiv_M b'$ such that $\langle a',b'\rangle\in R'$.
  Then $a'\in A$ and $b'\in A$ so $a=a'$ and $b=b'$.
\end{proof}

\begin{fact}\label{fact_reduced_funct} 
  Every reduced p-morphism $R$ is (the graph of) a map $f:{\rm dom}\,R\to{\rm rng}\,R$. Therefore every reduced p-isomorphism is a bijection between its domain and range.
\end{fact}

\begin{proof}
  Suppose not, then there are $a,b,c$ such that $\langle a,b\rangle,\langle a,c\rangle\in R$ but $b\neq c$.
  As $R$ is reduced, then $b\not\equiv_M c$.
  But this  is not possible since $\equiv_M$ is definable by a p-type.
  The second claim follows immediately.
\end{proof}

\begin{fact}[ ($\infty$-p-homogeneity)]\label{fact_phogeneity}
  Every p-morphism extends to a p-morphism defined on the whole of $\U^x$.
  Every p-isomorphism extends to a p-automorphism of $\U^x$.
\end{fact}
\begin{proof}
  Let $R$ be a p-morphism.
  Pick be a reduced p-morphism $f$ such that $R\ \subseteq\ \mathrel{\equiv_M}\circ \mathrel{f}\circ\mathrel{\equiv_M}$.
  Let $\bar a$ be an enumeration of ${\rm dom} f$ and let $b\in\U^x$.
  Let $p(\bar x)=\textrm{p-tp}(\bar a)$ and let $q(\bar x,y)=\textrm{p-tp}(\bar a,b)$.
  By $\infty$-p-saturation $\exists y\,q(\bar x,y)$ is equivalent to the type containing the existential closure of the formulas in $q(\bar x,y)$.
  By Fact~\ref{fact_p_qe}, the latter is equivalent to a p-type, which is necessarily coicides with $p(\bar x)$.
  Then $q(f\bar a,y)$ is consistent hence by $\infty$-p-saturation, $f$ can be extended to a p-morphism defined on $b$.

  The second claim follows by back-and-forth.
\end{proof}

A set $\A\subseteq\U^x$ is \emph{p-maximal} if it satisfies the following (obviously) equivalent conditions for every $\bar a\in\A^{\bar x}$, where $\bar x$ is a finite tuple, and $\bar b$ such that $\langle\bar a,\bar b\rangle\in R$ for some p-morphism $R$

\begin{itemize}
  \item[a.] $r_{\phi\,;\alpha}(\bar a)\ \ \Leftrightarrow\ \ r_{\phi\,;\alpha}(\bar b)$ \hfill
  for every $r_{\phi\,;\alpha}\in L^{\rm\! p}$

\item[b.] $\textrm{p-tp}(\bar a)\ \ =\ \ \textrm{p-tp}(\bar b)$
% \item[3.] $\textrm{p-tp}(\bar a)$ is a maximally consistent p-type.
\end{itemize}

% By Zorn's lemma, every set satisfying the above conditions is contained in a p-maximal set.

\begin{fact}\label{fact_pmaximal}
  Then the following are equivalent
  \begin{itemize}
    \item[1.] $\A\subseteq\U^x$ is p-maximal
    \item[2.] $\textrm{p-tp}(\bar a)$ is a maximally consistent p-type for every $\bar a\in\A^{\bar x}$ and every finite $\bar x$
    \item[3.] $\textrm{p-tp}(\bar c)$ is a maximally consistent p-type for $\bar c$ any enumeration of $\A$
    \item[4.] every p-morphism defined on $\A$ is a p-isomorphism between $\A$ and its image.
  \end{itemize}
\end{fact}

\begin{proof}
  1$\Rightarrow$2. Negate 2. Suppose $\textrm{p-tp}(\bar a)$ is not maximal for some finite $\bar a\in\A^{\bar x}$.
  Pick some $\bar a'\in\A^{\bar x}$ such that $\textrm{p-tp}(\bar a)\subset \textrm{p-tp}(\bar a')$.
  Then $R=\{\langle \bar a,\bar a'\rangle\}$ is a p-morphism that contradicts the p-maximality of $\A$.
  
  2$\Rightarrow$3. If $\bar c$ is an enumeration of $\A$ that witnesses the failure of 3, then some finite subtuple of $\bar c$ witnesses the failure of 2.

  3$\Rightarrow$4. Negate 4. Let Let $R$ be ba p-morphism defined on $\A$ that is not a p-isomorphism.
  Let $\bar a\in\A^{\bar x}$ and $\bar b$ be such that $\langle \bar a,\bar b\rangle\in R$ and $\textrm{p-tp}(\bar a)\smallsetminus \textrm{p-tp}(\bar b)\neq\varnothing$.
  To contradict 3, it suffices to extend $\{\langle \bar a,\bar b\rangle\}$ to a p-morphism defined on $\A$.
  This is possible by $\infty$-p-homogeneity.

  4$\Rightarrow$1. Clear.
\end{proof}

\begin{fact}
  Let $\J\subseteq\U^x$ be a p-maximal subsets of $\U^x$.
  Then ${\equiv_M}[\J]$, the image of $\J$ under the p-morphism $\equiv_M$, is maximal with respect to inclusion among the p-maximal subsets of $\U^x$.
\end{fact}

\begin{proof}
  Suppose for a contradiction that ${\equiv_M}[\J]\subset\A$, where $\A$ is a p-maximal set.
  Pick any $a\in\A\smallsetminus{\equiv_M}[\J]$.
  
  and let $b\in\J$ be such that $a\equiv_M b$.
  Then there is a p-morphism $R$ such that ${\equiv_M}[\J]=R[\J]$.
  
  $\A$ be a p-maximal set and let $R$ be a p-morphism of $\J$ into $\A$.
  By Fact~\ref{fact_pmaximal}, $R$ is an isomorphism between $\J$ and its image $R[\J]$, which is also p-maximal.
  By $\infty$-p-homogeneity, $R^{-1}$ extends to a p-morphism $R'$ defined on $\A$.
  Then $R'[\A]$ is a p-maximal set that contains $\J$.
  By the maximality of $\J$ under inclusion, we have $\J = R'[\A]$.
\end{proof}

A \emph{p-retraction} of $\U^x$ onto $\A$ is a p-morphism that maps $\U^x$ onto $\A$ and is the identity on $\A$.

\begin{theorem}
  All p-maximal subsets of $\U^x$ are p-isomorphic, and for any p-maximal set $\A$ there is a p-retraction of $\U^x$ onto $\B$.
\end{theorem}

\begin{proof}
  Let $\bar c$ be an enumeration of $\U^x$ and let $\bar a$ be a tuple such that $\textrm{p-tp}(\bar c)\subseteq \textrm{p-tp}(\bar a)$ is p-maximal.
  Then $f=\{\langle\bar c,\bar a\rangle\}$ is a p-morphism onto the p-maximal set $\A={\rm rng}\,f$.
  Let $\B\subseteq\U^x$ be any other p-maximal set.
  By 4 in Fact~\ref{fact_pmaximal}, $f$ is an isomorphism between $\B$ and $\A$.
  By the same argument $f\restriction\A$ is a p-automorphism of $\A$.
  Then $(f\restriction\A)^{-1}\circ f$ is a p-retraction of $\U^x$ onto $\A$.
\end{proof}
