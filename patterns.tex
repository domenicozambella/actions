\section{Patterns}
\label{patterns}

\def\ceq#1#2#3{\noindent\parbox[t]{22ex}{$\displaystyle #1$}\parbox{6ex}{\hfil $#2$}{$\displaystyle #3$}}

In the following $\U$ is some monster model of a complete theory $T$ of language $L$.
For simplicity $T$ is assumed to be one-sorted.

We define a structure with domain $\U^x$, where $x$ is some fixed tuple of variables.
The language is denoted by $L^{\rm\! p}$.
Though $L^{\rm\! p}$ depends on $x$, this will be left implicit.

Let $V$ be some large set of variables that are copies of $x$. 
Let $\Delta$ be the set of positive Boolean combinations of formulas in $\ \bigcup_{x'\in V} L_{x'\,;z}$, where $z=\langle z_i:i\in\omega\rangle$ is a fixed tuple of variables of length $\omega$.
The choice of $\Delta$ is justified in the remark below.

For every formula $\phi(x_0,\dots,x_{n-1}\,;z)\in\Delta$ and every type $\alpha(z)\subseteq L$ the language $L^{\rm\! p}$ contains the $n$-ary relation \emph{$r_{\phi\,;\alpha}$.}
We require that $\alpha(z)$ is a finitary type, i.e.\@ only a finite number of variables actually occur in it.

Below by the symbol \emph{$L^{\rm\! p}$} we will also denote the set of \textit{atomic\/} formulas $r_{\phi\,;\alpha}(x_0,\dots,x_{n-1})$ for any $x_0,\dots,x_{n-1}\in V$.
Note that equalities are \textit{not\/} included in $L^{\rm\! p}$.

For a given model weakly saturated model $M$, we define the interpretation of $r_{\phi\,;\alpha}$ in $\U^x$.
Eventually, we will proved that the main definition in this section, the definition of \textit{core,} does not depend on the choice of $M$.
However, some of the intermediate notions introduced here do depend on $M$, though to avoid cluttering the notation we will omit reference to it.

If $\bar a=a_0,\dots,a_{n-1}\in\U^x$, we declare $r_{\phi\,;\alpha}(\bar a)$ true if $\phi(\bar a\,;b)$ holds for every $b\in M^z$ such that $\alpha(b)$.
We repeat the definition for emphasis

\ceq{\hfill r_{\phi\,;\alpha}(\bar a)}{\Leftrightarrow}{\forall z\in\alpha(M^z)\ \phi(\bar a\,;z).}

The following is paramount.

\begin{remark}
  By the choice of $\Delta$, the truth of $r_{\phi\,;\alpha}(\bar a)$ only depends on the equivalence class of $\equiv_M$ of the components of the tuple $\bar a$.
  Explicitely, if $\bar a=a_0,\dots,a_{n-1}$ and $\bar a'=a'_0,\dots,a'_{n-1}$ are such that $a_i\equiv_M a'_i$ for every $i<n$, then $r_{\phi\,;\alpha}(\bar a)\ \leftrightarrow\ r_{\phi\,;\alpha}(\bar a')$.
\end{remark}

By the remark, it would be natural to define the interpretation of $L^{\rm\! p}$ on the quotient $\U^x/{\equiv_M}$ or on the type space $S_x(M)$, but we prefer to keep the original domain $\U^x$ to have a common playground with $L$.

A \emph{p-type\/} is a set of (atomic) formulas in $L^{\rm\! p}(\U^x)$.

Note incidentally that the relation $x_0\equiv_M x_1$ is definable by the p-type containing the formulas $r_{\theta\,; \top }(x_0,x_1)$ for every $\theta(x_0,x_1\,;z)$ of the form $\phi(x_0\,;z)\leftrightarrow\phi(x_1\,;z)$.

We write \emph{$\textrm{p-tp}(\bar a)$} for the p-type of $\bar a$, i.e.\@ the set of formulas $r_{\phi\,;\alpha}(\bar x)$ such that $r_{\phi\,;\alpha}(\bar a)$ holds.

\begin{fact}\label{fact_p_qe}
  Formulas in $L^{\rm\! p}$ are closed under adjunction of dummy variables, disjunction, conjunction, and existential quantification.
\end{fact}

\begin{proof}
  The claim about dummy variables is clear, so we tacitatly use it below.

  Up to renaming of the variables that occur in $\beta(z)$ and $\psi(\bar x\,;z)$, we can assume that they are disjoint from those in $\alpha(z)$.
  Therefore the following equivalences hold 
  
  \ceq{\hfill r_{\phi\,;\alpha}(\bar a)\vee r_{\psi\,;\beta}(\bar a)}{\Leftrightarrow}{\forall z\in\alpha(M^z)\ \phi(\bar a\,;z)\ \ \vee\ \ \forall z\in\beta(M^z)\ \psi(\bar a\,;z)} 

  \ceq{}{\Leftrightarrow}{\forall z\in\alpha(M^z)\cup\beta(M^z)\ \ \ \phi(\bar a\,;z)\, \vee\,\psi(\bar a\,;z)} 

  \ceq{}{\Leftrightarrow}{r_{\phi\vee\psi\,;\,\alpha\cup\beta}(\bar a)}


  A similar argument applies to the conjunction of two formulas.

  Finally, consider the formula $\exists y\,r_{\phi\,;\alpha}(\bar x,y)$.
  Without loss of generality, we can assume that $\phi$ has the form

  \ceq{\hfill\phi(\bar x,y\,;z)}{=}{\bigvee_{i=1}^m\phi_i(\bar x,y\,;z)}

  where $\phi_i(\bar x,y\,;z)\,=\,\phi'_i(\bar x\,;z)\,\wedge\,\psi_i(y\,;z)$ and each $\phi'_i(\bar x\,;z)$ is a conjunction of formulas in $\bigcup_{x'\in V} L_{x'\,;z}$.
  Closure under existential quantification follows from the following equivalences

  \ceq{\hfill\exists y\ r_{\phi\,;\alpha}(\bar a, y)}{\Leftrightarrow}{\exists y\,\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^m\phi_i(\bar a,y\,;z)\Big]}

  \ceq{}{\Leftrightarrow}{\forall z\Big[ z\in\alpha(M)\rightarrow \bigvee_{i=1}^m\phi'_i(\bar a\,;z)\wedge\exists y\,\psi_i(y\,;z)\Big]}

  \ceq{}{\Leftrightarrow}{\vphantom{\bigvee_{i=1}^n}r_{\theta\,;\alpha}(\bar a),}

  where 
  
  \ceq{\hfill\theta(\bar x\,;z)}{=}{\bigwedge_{i=1}^m\phi'_i(\bar x\,;z)\ \wedge\ \exists y\,\psi_i(y\,;z)}

  is a positive Boolean combination of formulas in $\ \bigcup_{x'\in V} L_{x'\,;z}$, as required.
\end{proof}


\begin{fact}[ ($\infty$-p-saturation)]\label{fact_psat}
  Let $\A\subseteq\U^x$ have arbitrary cardinality.
  Let $\bar x$ be a tuple of copies of $x$ of length $\le|\U|$.
  Then every p-type $p(\bar x)\subseteq L^{\rm\! p}(\A)$ that is finitely consistent in $\U^x$ is realized in $\U^x$.
\end{fact}

\begin{proof}
  Let $A\subseteq\U^x$ be a small set containing a representative of every $\equiv_M$ equivalence class that intersects $\A$.
  Replacing $\A$ in $p(\bar x)$ by elements $A$ we obtain a equivalent type $p'(\bar x)\subseteq L^{\rm\! p}(A)$.
  Note that every formula $r_{\phi\,;\alpha}(\bar x)$ there is a type $q(\bar x)\subseteq L$ such that

  \ceq{\hfill\U^x\models r_{\phi\,;\alpha}(\bar a)}{\Leftrightarrow}{\U\models q(\bar a)}

  Therefore the saturation of $\U$ implies that $p'(\bar x)$ is realized in $\U^x$.
\end{proof}

\section{Morphism}

As the interpretation of $L^{\rm p}$ factors through the equivalence relation $\equiv_M$, p-morphims could be introduced as (partial) maps of $\U^x/{\equiv_M}$ into itself.
However there are a few occasions when it is preferable to work with maps of $\U^x$ into itself. 
Though the switching between the two perspectives is intuitively obvious, it requires a fair amount of notation.

Let $R\mathrel{\subseteq}\U^x\times\U^x$ be a binary relation.
If $\bar a=\langle a_i:i<\lambda\rangle$ and $\bar b=\langle b_i:i<\lambda\rangle$ we write $\langle \bar a,\bar b\rangle\in R$ to abbreviate: $\langle a_i,b_i\rangle\in R$ for all $i<\lambda$.
We say that $R$ is a \emph{p-morphism} if the following condition holds 

\ceq{\#\hfill\U^x\models r_{\phi\,;\alpha}(\bar a)}{\Rightarrow}{\U^x\models r_{\phi\,;\alpha}(\bar b)}\hfill for every $r_{\phi\,;\alpha}\in L^{\rm p}$ and every $\langle \bar a,\bar b\rangle\in R$.

When the converse implication in \# also holds, i.e.\@ when also $R^{-1}$ is a p-morphism, we say that $R$ is a \emph{partial p-isomorphism.}
When $\A\subseteq{\rm dom}\,R$,  $\B\subseteq{\rm range}\,R$, and $R\cap (\A\times\B)$ is a partial p-isomorphism, we say that $R$ is an \emph{p-isomorphism} between $\A$ and $\B$ or, when $\A=\B$, a \emph{p-automorphism} of $\A$.

If $R$ is a p-morphism, also $\mathrel{\equiv_M}\circ \mathrel{R}\circ\mathrel{\equiv_M}$ is a p-morphism.
This we call the \emph{full} p-morphism generated by $R$.
We say that $R$ and $R'$ are \emph{equivalent} if they generate the same full p-morphism.

\begin{remark}
  We can identify p-morphisms with maps $f:\U^x/{\equiv_M}\to\U^x/{\equiv_M}$.
  In fact, to any such map we can associate the relation $R=\bigcup_{a\in\U^x}\big((a/{\equiv_M})\times f(a/{\equiv_M})\big)$.\smallskip

  Vice versa, to any p-morphisms $R$ we can associate $f:\U^x/{\equiv_M}\to\U^x/{\equiv_M}$ that maps $a/{\equiv_M}\mapsto b/{\equiv_M}$ for some/any $b$ such that $\langle a,b\rangle\in R$.
  This is well-defined because if $\langle a,b\rangle,\langle a',b'\rangle\in R$  and $a\equiv_M a'$, then we necessarily have $b\equiv_M b'$.\smallskip

  We find more convenient to work with relations. 
\end{remark}


% Note that $R\circ R^{-1}$ and $R^{-1}\!\circ R$ are contained in the (trivial) p-isomorphism $\equiv_M$.

We say that the p-morphism $R$ is \emph{reduced\/} if there is no $R'\ \subset\ R$ that is equivalent to $R$.
It is clear that if $A, B\subseteq\U^x$ are any two sets of represetative of $\equiv_M$ then $f=R\cap (A\times B)$ is a reduced morphism equivalent to $R$.

\begin{fact}\label{fact_reduced_funct}
  Any reduced p-morphism is the graph of a (partial) map.
  Therefore every reduced p-isomorphism is a bijection.
\end{fact}

\begin{proof}
  Suppose not, then there are $a,b,c$ such that $\langle a,b\rangle,\langle a,b'\rangle\in R$ but $b\neq b'$.
  As $R$ is reduced, then $b\not\equiv_M b'$.
  As noted above, this  is not possible.
  The second claim follows immediately.
\end{proof}

\begin{fact}\label{fact_phogeneity}
  Every p-morphism extends to a p-morphism defined on the whole of $\U^x$.
  Also, every p-isomorphism extends to a p-automorphism of $\U^x$.
\end{fact}
\begin{proof}
  Let $R$ be a p-morphism.
  Pick be a reduced p-morphism $f$ equivalent to $R$.
  Let $\bar a$ be an enumeration of ${\rm dom} f$ and let $b\in\U^x$.
  Let $p(\bar x)=\textrm{p-tp}(\bar a)$ and let $q(\bar x,y)=\textrm{p-tp}(\bar a,b)$.
  By $\infty$-p-saturation $\exists y\,q(\bar x,y)$ is equivalent to the type containing the existential closure of the formulas in $q(\bar x,y)$.
  By Fact~\ref{fact_p_qe}, the latter is equivalent to a p-type, which is necessarily coicides with $p(\bar x)$.
  Then $q(f\bar a,y)$ is consistent hence by $\infty$-p-saturation it is realized by some $c\in\U^x$.
  Then $f\cup\{\langle b,c\rangle\}$ is a p-morphism defined in $b$ that extends $f$.
  Then $R$ can be extended as well.

  The second claim is obtained by back-and-forth.
\end{proof}

Let $m$ be an enumeration of $M$.

\begin{remark}
  We remark an simple fact which is used in the proof below. 
  We say that $\phi'(a\,;z)$ is an alphabetic variant of $\phi(a\,;z)$ if it is obtained by replacing the vartiables effectively occurring in $\phi(a\,;z)$ by other variables (among those in $z$).
  The following are clearly equivalent
  \begin{itemize}
    \item[1.] $\phi(a\,;M^z)=M^z$
    \item[2.] $\phi'(a\,;m)$ holds for every alphabetic variant of $\phi(a\,;z)$.
  \end{itemize}
\end{remark}

Let $S_{m,M}\subseteq S_z(\U)$, where $z$ has the  length of $m$, be the set of global types that extend ${\rm tp}(m/\varnothing)$ and are finitely satisfied in $M$. 

\begin{fact}
  Assume $x$ has infinite length.
  Then there is a one-to-one correspondence between the elements of $S_{m,M}$ and the total p-morphisms of $\U^x$ up to equivalence.
\end{fact}

\begin{proof}
  As $x$ has infinite length, we can assume that the formulas in the types in $S_z(\U)$ are the form $\phi(a\,;z)$ for some $\phi(x\,;z)\in L$ and $a\in\U^x$.

  Let $R\subseteq\U^x\times\U^x$ be a total p-morphism.
  Let $p(z)$ be the type containing the formulas $
  \phi(a\,;z)$ such that $\phi(b\,;m)$ for some/any $b$ such that $\langle a,b\rangle\in R$.
  We claim that $p(z)\in S_{m,M}$.
  It easy to see that $p(z)$ is complete.
  The inclusion $p(z)\supseteq{\rm tp}(m/\varnothing)$ is clear, so we only need to prove that $p(z)$ is finitely satisfied in $M$.
  Suppose not, then $\psi(a_1,\dots,a_n\,;M^z)=\varnothing$, where 
  
  \ceq{\hfill\psi(x_1,\dots,x_n\,;z)}{=}{\bigwedge_{i=n}^n\phi_i(x_i\,;z)}
  
  for some $\phi_i(x\,;z)\in L$ and $\langle a_i,b_i\rangle\in R$ such that $\phi_i(b_i\,;m)$.
  Then $r_{\neg\psi\,;\top}(a_1,\dots,a_n)$ and $\neg r_{\neg\psi\,;\top}(fa_1,\dots,fa_n)$.
  A contradiction.

  Vice versa, let $p(z)\in S_{m,M}$ be given.
  Let $R\subseteq\U^x\times\U^x$ contain the pairs $\langle a,b\rangle$ such that $\phi(a\,;z)\in p\Leftrightarrow\phi(b\,;m)$ holds for every $\phi(x\,;z)\in L$.
  We prove that $R$ is a p-morphism.

  Let $\psi(x_1,\dots,x_n\,;z)\in\Delta$, $\alpha(z)\subseteq L$, and $a_1,\dots,a_n\in\U^x$ be such that $r_{\psi\,;\alpha}(a_1,\dots,a_n)$.
  Finally, let $\langle a_i,b_i\rangle\in R$.
  We can assume $\alpha(M^z)\neq\varnothing$, otherwise $r_{\psi\,;\alpha}(b_1,\dots,b_n)$ trivially holds.
  Assume

  \ceq{\hfill\psi(x_1,\dots,x_n\,;z)}{=}{\bigvee_{h=1}^m\bigwedge_{k=1}^m\phi_{h,k}(x_{i_{h,k}}\,;z)}

  for some $i_{h,k}\in\{1,\dots,n\}$.

  As $r_{\psi\,;\alpha}(a_1,\dots,a_n)$, up to re-indexing, we obtain that $\phi_{0,k}(a_{i_{0,k}}\,;c)$ every $c\in\alpha(M^z)$.
  As $p(z)$ is finitely satisfiable in $M$, then $\alpha(z)\wedge\neg\phi_{0,k}(a_{i_k}\,;z)\notin p$.
  Then, by completeness, $p(z)\vdash\alpha(z)\rightarrow\phi_{0,k}(b_{i_k}\,;z)$.
  Then $m\models\alpha(z)\rightarrow\phi_{0,k}(b_{i_k}\,;z)$.

\end{proof}

A set $\A\subseteq\U^x$ is \emph{p-maximal} if it satisfies the following (obviously) equivalent conditions for every $\bar a\in\A^{\bar x}$, where $\bar x$ is a finite tuple, and $\bar b$ such that $\langle\bar a,\bar b\rangle\in R$ for some p-morphism $R$

\begin{itemize}
  \item[a.] $r_{\phi\,;\alpha}(\bar a)\ \ \Leftrightarrow\ \ r_{\phi\,;\alpha}(\bar b)$ \hfill
  for every $r_{\phi\,;\alpha}\in L^{\rm\! p}$

\item[b.] $\textrm{p-tp}(\bar a)\ \ =\ \ \textrm{p-tp}(\bar b)$
% \item[3.] $\textrm{p-tp}(\bar a)$ is a maximally consistent p-type.
\end{itemize}

% By Zorn's lemma, every set satisfying the above conditions is contained in a p-maximal set.

\begin{fact}\label{fact_pmaximal}
  The following are equivalent
  \begin{itemize}
    \item[1.] $\A\subseteq\U^x$ is p-maximal
    \item[2.] $p(\bar x)=\textrm{p-tp}(\bar a)$ is a maximally consistent p-type for every $\bar a\in\A^{\bar x}$ and every finite $\bar x$
    \item[3.] $p(\bar y)=\textrm{p-tp}(\bar c)$ is a maximally consistent p-type for $\bar c$ any enumeration of $\A$
    \item[4.] every p-morphism defined on $\A$ is a p-isomorphism between $\A$ and its image.
  \end{itemize}
\end{fact}

\begin{proof}
  1$\Rightarrow$2. Negate 2. Suppose $\textrm{p-tp}(\bar a)$ is not maximal for some finite $\bar a\in\A^{\bar x}$.
  Pick some $\bar a'\in\A^{\bar x}$ such that $\textrm{p-tp}(\bar a)\subset \textrm{p-tp}(\bar a')$.
  Then $R=\{\langle \bar a,\bar a'\rangle\}$ is a p-morphism that contradicts the p-maximality of $\A$.
  
  2$\Rightarrow$3. If $\bar c$ is an enumeration of $\A$ that witnesses the failure of 3, then some finite subtuple of $\bar c$ witnesses the failure of 2.

  3$\Rightarrow$4. Negate 4. Let Let $R$ be ba p-morphism defined on $\A$ that is not a p-isomorphism.
  Let $\bar a\in\A^{\bar x}$ and $\bar b$ be such that $\langle \bar a,\bar b\rangle\in R$ and $\textrm{p-tp}(\bar a)\smallsetminus \textrm{p-tp}(\bar b)\neq\varnothing$.
  To contradict 3, it suffices to extend $\{\langle \bar a,\bar b\rangle\}$ to a p-morphism defined on $\A$.
  This is possible by $\infty$-p-homogeneity.

  4$\Rightarrow$1. Clear.
\end{proof}

\begin{fact}\label{fact_pmaximalset}
  If $\A$ is p-maximal, also $(\equiv_M)[\A]=\{a\ :\ \exists x\ a\equiv_Mx\in\A\}$ is p-maximal.
\end{fact}

\begin{proof}
  By 4 in Fact~\ref{fact_pmaximal}.
  Indeed, let $R$ be a p-morphism defined on $(\equiv_M)[\A]$.
  As $R\circ(\equiv_M)$ is also a p-morphism, $R\circ(\equiv_M)[\A]$ is p-maximal.
\end{proof}

A set $\J\subseteq\U^x$ is called a \emph{core} if it is p-maximal and, among the p-maximal sets, it is maximal w.r.t.\@ inclusion.

\begin{fact}\label{fact_onto}
  Let $\J$ be a core.
  Let $\A$ be a p-maximal set.
  Then every p-morphism $R\subseteq\A\times\J$ extends to a p-morphism $R'$ such that $\A={\rm dom}R'$ and ${\rm range}R'\subseteq\J$. 
\end{fact}

\begin{proof}
  Reason as in the proof of Fact~\ref{fact_phogeneity} with the same notation but choose $b\in\A$.
  Let $R'$ be a p-morphism defined on $\A$ that extends $f\cup\{\langle b,c\rangle\}$.
  Then $R'[\A]$, i.e.\@ the image of $\A$ under $R'$, is a p-maximal set that contains $\J$.
  By maximality $R'[\A]$ coincides with $\J$, hence $c\in\J$.
\end{proof}

By back-and-forth we obtain the following.

\begin{corollary}[ ($\infty$-p-homogeneity of the core)]\label{fact_corephogeneity}
  If $\J$ and $\J'$ are cores, then every p-morphism $R\subseteq\J\times\J'$ extends to a p-isomorphism between $\J$ and $\J'$.
\end{corollary}

A \emph{p-retraction} of $\U^x$ onto $\A$ is a total p-morphism $R$ that coincides with $\equiv_M$ on $\A$, i.e.\@ such that $R\cap (\A\times\A)\ \ =\ \ ({\equiv_M})\cap (\A\times\A)$.

\begin{theorem}
  If $\J$ and is a core then there is a p-retraction of $\U^x$ onto $\J$
\end{theorem}

\begin{proof}

\end{proof}
